CREATE DATABASE aw_sampling WITH ENCODING='UTF8' OWNER=postgres TEMPLATE=postgis_template CONNECTION LIMIT=-1;  

# Set grants
GRANT ALL ON DATABASE MapWinGIS TO postgres;
